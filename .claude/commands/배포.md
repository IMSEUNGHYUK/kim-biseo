# 배포 명령어

마케팅팀장 대시보드를 GitHub & Vercel에 배포합니다.

```bash
#!/bin/bash

# 1️⃣ 변경된 파일 확인
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "📋 변경된 파일 확인 중..."
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
git status

CHANGES=$(git diff --name-only)
UNTRACKED=$(git ls-files --others --exclude-standard)

if [ -z "$CHANGES" ] && [ -z "$UNTRACKED" ]; then
  echo "✨ 변경된 파일이 없습니다"
  exit 0
fi

echo ""
echo "📝 변경/추가된 파일:"
echo "$CHANGES"
echo "$UNTRACKED"

# 2️⃣ 커밋 메시지 자동 생성
echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "✍️  커밋 메시지 생성 중..."
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

FILE_COUNT=$(echo "$CHANGES" "$UNTRACKED" | wc -l)

if echo "$CHANGES" | grep -q "업무대시보드.html"; then
  COMMIT_MSG="Update: 업무대시보드 개선 및 기능 추가"
elif echo "$CHANGES" | grep -q "매출"; then
  COMMIT_MSG="Add: 매출 분석 리포트 추가"
elif echo "$CHANGES" | grep -q "분석보고서"; then
  COMMIT_MSG="Add: 사이트 분석 보고서 추가"
else
  COMMIT_MSG="Update: 프로젝트 파일 업데이트 ($FILE_COUNT개 파일)"
fi

echo "💭 생성된 메시지: $COMMIT_MSG"

# 3️⃣ GitHub에 푸쉬
echo ""
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "🚀 GitHub에 배포 중..."
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"

git add .
git commit -m "$COMMIT_MSG"
PUSH_RESULT=$(git push origin main 2>&1)

if [ $? -eq 0 ]; then
  echo "$PUSH_RESULT"
  
  # 4️⃣ 배포 완료 메시지
  echo ""
  echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
  echo "✅ 배포 완료!"
  echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
  echo ""
  echo "📌 커밋 메시지: $COMMIT_MSG"
  echo "🔗 GitHub: https://github.com/IMSEUNGHYUK/kim-biseo"
  echo "🌐 Vercel: https://kim-biseo.vercel.app"
  echo ""
  echo "💡 팁: Vercel 대시보드에서 배포 상태를 확인할 수 있습니다"
  echo ""
else
  echo "❌ 푸시 실패:"
  echo "$PUSH_RESULT"
  exit 1
fi
```

## 📌 사용법

```
/배포
```

## 🎯 동작 순서

1. **파일 변경 확인** — `git status` 출력
2. **자동 메시지 생성** — 변경 파일 기반으로 스마트 메시지
3. **전체 추가** — `git add .`
4. **커밋 및 푸시** — GitHub main 브랜치로 업로드
5. **배포 완료** — 성공 메시지 + 링크 표시

## ✨ 특징

- ⚡ **자동 커밋 메시지** — 변경 파일 분석해 적절한 메시지 생성
- 🎨 **보기 좋은 출력** — 이모지와 구분선으로 명확한 진행 상황
- 🔗 **빠른 링크** — 배포 후 GitHub/Vercel 링크 자동 출력
- 🛡️ **안전 체크** — 변경 없으면 조용히 종료
